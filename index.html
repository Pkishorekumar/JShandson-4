11:-Birthday Game

      
    function Birthday_Game(arr,D ,M) {
    
        let a =0,b;
    
    if(arr.length< M)
    return a;
    
    for(let i=0;i< arr.length;i++)
    {
    b=arr.slice(i,i+M);
    
    if(b.reduce((acc, cur)=> acc + cur)=== D)
    a++;
    }
    return a;
    }
